<!DOCTYPE html>

<html>

<head>

  <meta charset="UTF-8" />

  <title>Noti</title>

</head>

<body>

  <script>

    // redirect to job if token exists

    const params = new URLSearchParams(window.location.search);


    const t = params.get("t");

    const id = params.get("id") || params.get("jobId");


    if (t && id) {

      window.location.href = `/job.html?t=${t}&id=${id}`;

    } else {

      document.body.innerHTML = "Noti is running 👍";

    }

  </script>

</body>

</html>

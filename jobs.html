  <script>

    const API_URL = "https://script.google.com/macros/s/AKfycbwqJ99QUqKPLwvejodDYqpz-0F2740qYAkiwHrP3h_aNHoDKcUTUIGvsRrDb1H17cTR/exec";


    const app = document.getElementById("app");

    const heroTitle = document.getElementById("heroTitle");

    const heroText = document.getElementById("heroText");

    const heroPills = document.getElementById("heroPills");


    const pathParts = window.location.pathname.split("/").filter(Boolean);

    const jobId = pathParts[pathParts.length - 1] || new URLSearchParams(window.location.search).get("id") || "";

    const token = new URLSearchParams(window.location.search).get("t") || "";


    let currentData = null;


    function escapeHtml(value) {

      return String(value || "")

        .replace(/&/g, "&amp;")

        .replace(/</g, "&lt;")

        .replace(/>/g, "&gt;")

        .replace(/\"/g, "&quot;")

        .replace(/'/g, "&#039;");

    }


    function pill(text, extra = "") {

      return `<div class="pill ${extra}">${escapeHtml(text)}</div>`;

    }


    async function fetchJob() {

      if (!jobId || !token) {

        renderError("This job link is missing information.");

        return;

      }


      try {

        const url = `${API_URL}?type=get_job&id=${encodeURIComponent(jobId)}&t=${encodeURIComponent(token)}`;

        const res = await fetch(url);

        const data = await res.json();


        if (!data.success) {

          renderError(data.message || "This job link is invalid.");

          return;

        }


        currentData = data;

        renderPage(data);

      } catch (err) {

        renderError("We could not load this job right now.");

      }

    }


    async function postAction(payload) {

      const res = await fetch(API_URL, {

        method: "POST",

        headers: {

          "Content-Type": "application/json"

        },

        body: JSON.stringify({

          type: "frontend_action",

          jobId,

          token,

          ...payload

        })

      });


      return res.json();

    }


    function setHero(data) {

      const role = data.role === "tradie" ? "Tradie" : "Customer";

      heroTitle.textContent = data.job.description || "Job";

      heroText.textContent = `${role} view for ${data.job.suburb || "your area"}.`;


      heroPills.innerHTML = [

        pill(data.role.toUpperCase()),

        pill(data.job.job_type || "STANDARD", "orange"),

        pill(data.stage || "LIVE")

      ].join("");

    }


    function renderError(message) {

      heroTitle.textContent = "Job unavailable";

      heroText.textContent = "This link may be invalid or expired.";

      heroPills.innerHTML = "";

      app.innerHTML = `

        <div class="card">

          <div class="card-body">

            <div class="notice error">${escapeHtml(message)}</div>

          </div>

        </div>

      `;

    }


    function renderPage(data) {

      setHero(data);


      if (data.role === "tradie_locked") {

        app.innerHTML = `

          <div class="card">

            <div class="card-body">

              <h2>Job already taken</h2>

              <p class="subtext">This job has already been locked in by another tradie.</p>

            </div>

          </div>

        `;

        return;

      }


      if (data.role === "tradie") {

        renderTradieView(data);

        return;

      }


      renderCustomerView(data);

    }

        <div class="card">

        ${searching ? `

          <div class="card">

            <div class="card-body">

              <div class="offer-card" style="margin-top:14px;">

                <h3>${escapeHtml(tradie ? tradie.name : "Tradie available")}</h3>

                <div class="meta">

                  <div><strong>ETA:</strong> ${escapeHtml(job.eta || "ASAP")}</div>

                  <div><strong>Price:</strong> ${escapeHtml(job.price ? "$" + job.price : "TBC")}</div>

                  ${tradie ? `<div class="muted">Licensed: ${escapeHtml(tradie.licence_verified || "")}, insured: ${escapeHtml(tradie.insurance_verified || "")}</div>` : ""}

                </div>

                <div class="actions">

                  <button class="btn-primary" id="customerAcceptBtn">Accept tradie</button>

                  <button class="btn-secondary" id="customerNextBtn">See next option</button>

                </div>

              </div>

            </div>

          </div>

        ` : ""}


        ${accepted && tradie ? `

          <div class="card">

            <div class="card-body">

              <h2>Tradie confirmed</h2>

              <div class="summary-card" style="margin-top:14px;">

                <h3>${escapeHtml(tradie.name)}</h3>

                <div class="meta">

                  <div><strong>Phone:</strong> ${escapeHtml(tradie.phone || "Shared once confirmed")}</div>

                  <div><strong>ETA:</strong> ${escapeHtml(job.eta || "TBC")}</div>

                  <div><strong>Price:</strong> ${escapeHtml(job.price ? "$" + job.price : "TBC")}</div>

                </div>

              </div>

            </div>

          </div>

        ` : ""}


        ${canReview ? `

          <div class="card">

            <div class="card-body">

              <h2>How did it go?</h2>

              <p class="subtext">Leave a quick review for this job.</p>

              <div class="actions">

                <button class="btn-primary" id="goReviewBtn">Leave review</button>

              </div>

            </div>

          </div>

        ` : ""}

      `;


      const notice = document.getElementById("customerNotice");

      const saveBtn = document.getElementById("customerSaveBtn");

      const submitBtn = document.getElementById("customerSubmitBtn");

      const acceptBtn = document.getElementById("customerAcceptBtn");

      const nextBtn = document.getElementById("customerNextBtn");

          try {

          } catch (err) {

          }

        });

      }


      if (submitBtn) {

        submitBtn.addEventListener("click", async () => {

            submitBtn.disabled = true;

            await postAction({

              action: "update_job",

              description: document.getElementById("cDescription").value,

              suburb: document.getElementById("cSuburb").value,

              postcode: document.getElementById("cPostcode").value,

          } catch (err) {

            setNotice(notice, err.message || "Could not submit job.", "error");

          }

          }

        });

            fetchJob();

          } catch (err) {

            alert(err.message || "Could not load next option.");

          } finally {

            nextBtn.disabled = false;

          }

        });

      }


      if (goReviewBtn) {

        goReviewBtn.addEventListener("click", () => {

          window.location.href = `/review/${encodeURIComponent(jobId)}?t=${encodeURIComponent(token)}`;

        });

      }

    }

  <script>

    const API_URL = "https://script.google.com/macros/s/AKfycbwqJ99QUqKPLwvejodDYqpz-0F2740qYAkiwHrP3h_aNHoDKcUTUIGvsRrDb1H17cTR/exec";

    const pathParts = window.location.pathname.split("/").filter(Boolean);

    const jobId = pathParts[pathParts.length - 1] || new URLSearchParams(window.location.search).get("id") || "";

    const token = new URLSearchParams(window.location.search).get("t") || "";


    const reviewTitle = document.getElementById("reviewTitle");

    const reviewText = document.getElementById("reviewText");

    const stars = document.getElementById("stars");

    const reviewComment = document.getElementById("reviewComment");

    const reviewNotice = document.getElementById("reviewNotice");

    const submitReviewBtn = document.getElementById("submitReviewBtn");


    let selectedRating = 0;


    function escapeHtml(value) {

      return String(value || "")

        .replace(/&/g, "&amp;")

        .replace(/</g, "&lt;")

        .replace(/>/g, "&gt;")

        .replace(/\"/g, "&quot;")

        .replace(/'/g, "&#039;");

    }


    function setNotice(text) {

      reviewNotice.innerHTML = text ? `<div class="notice">${escapeHtml(text)}</div>` : "";

    }


    function renderStars() {

      stars.innerHTML = "";

      for (let i = 1; i <= 5; i++) {

        const btn = document.createElement("button");

        btn.type = "button";

        btn.className = "star-btn" + (selectedRating === i ? " active" : "");

        btn.textContent = `${i} ★`;

        btn.addEventListener("click", () => {

          selectedRating = i;

          renderStars();

        });

        stars.appendChild(btn);

      }

    }


    async function fetchJob() {

      if (!jobId || !token) {

        reviewTitle.textContent = "Invalid review link";

        reviewText.textContent = "This review link is missing information.";

        submitReviewBtn.disabled = true;

        return;

      }


      try {

        const url = `${API_URL}?type=get_job&id=${encodeURIComponent(jobId)}&t=${encodeURIComponent(token)}`;

        const res = await fetch(url);

        const data = await res.json();


        if (!data.success) {

          reviewTitle.textContent = "Review unavailable";

          reviewText.textContent = data.message || "This review link is invalid.";

          submitReviewBtn.disabled = true;

          return;

        }


        reviewTitle.textContent = data.job.description || "Job review";

        reviewText.textContent = `Review for ${data.job.suburb || "your job"}.`;

      } catch (err) {

        reviewTitle.textContent = "Review unavailable";

        reviewText.textContent = "We could not load this review page right now.";

        submitReviewBtn.disabled = true;

      }

    }


    async function submitReview() {

      if (!selectedRating) {

        setNotice("Please choose a rating first.");

        return;

      }


      try {

        submitReviewBtn.disabled = true;

        setNotice("Submitting your review...");


        const res = await fetch(API_URL, {

          method: "POST",

          headers: {

            "Content-Type": "application/json"

          },

          body: JSON.stringify({

            type: "frontend_action",

            jobId,

            token,

            action: "rating",

            rating: selectedRating,

            comment: reviewComment.value

          })

        });


        const data = await res.json();

        if (!data.success) throw new Error(data.message || "Could not submit review.");


        setNotice("Thanks for your feedback.");

        submitReviewBtn.disabled = true;

        reviewComment.disabled = true;

        stars.innerHTML = "";

      } catch (err) {

        setNotice(err.message || "Could not submit review.");

        submitReviewBtn.disabled = false;

      }

    }


    submitReviewBtn.addEventListener("click", submitReview);


    renderStars();

    fetchJob();
